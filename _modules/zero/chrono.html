[docs]    def save(self, title):
        if self.is_enabled:
            now = datetime.now()
            delta = now - self.checkpoint
            logging.info('Chrono: %s [%dms]', title,
                         round(delta.total_seconds() * 1000))
            self.checkpoint = now
